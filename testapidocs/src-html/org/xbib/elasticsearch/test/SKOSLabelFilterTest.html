046    public class SKOSLabelFilterTest extends AbstractFilterTest {
047    
048        @BeforeMethod
049        @Override
050        public void setUp() throws Exception {
051            super.setUp();
052            skosAnalyzer = new SKOSAnalyzer(matchVersion, skosEngine,
053                    SKOSAnalyzer.ExpansionType.LABEL);
054            writer = new IndexWriter(directory, new IndexWriterConfig(matchVersion,
055                    skosAnalyzer));
056        }
057    
058        @Test
059        public void termQuerySearch() throws CorruptIndexException, IOException {
060    
061            Document doc = new Document();
062            doc.add(new Field("content", "The quick brown fox jumps over the lazy dog",
063                    TextField.TYPE_STORED));
064    
065            writer.addDocument(doc);
066    
067            searcher = new IndexSearcher(DirectoryReader.open(writer, false));
068    
069            TermQuery tq = new TermQuery(new Term("content", "hops"));
070    
071            Assert.assertEquals(1, TestUtil.hitCount(searcher, tq));
072    
073        }
074    
075        @Test
076        public void phraseQuerySearch() throws CorruptIndexException, IOException {
077    
078            Document doc = new Document();
079            doc.add(new Field("content", "The quick brown fox jumps over the lazy dog",
080                    TextField.TYPE_STORED));
081    
082            writer.addDocument(doc);
083    
084            searcher = new IndexSearcher(DirectoryReader.open(writer, false));
085    
086            PhraseQuery pq = new PhraseQuery();
087            pq.add(new Term("content", "fox"));
088            pq.add(new Term("content", "hops"));
089    
090            Assert.assertEquals(1, TestUtil.hitCount(searcher, pq));
091    
092        }
093    
094        @Test
095        public void queryParserSearch() throws IOException, QueryNodeException {
096    
097            Document doc = new Document();
098            doc.add(new Field("content", "The quick brown fox jumps over the lazy dog",
099                    TextField.TYPE_STORED));
100    
101            writer.addDocument(doc);
102    
103            searcher = new IndexSearcher(DirectoryReader.open(writer, false));
104    
105            Query query = new StandardQueryParser(skosAnalyzer).parse("\"fox jumps\"", "content");
106    
107            Assert.assertEquals(1, TestUtil.hitCount(searcher, query));
108    
109            Assert.assertEquals("content:\"fox (jumps hops leaps)\"", query.toString());
110            Assert.assertEquals("org.apache.lucene.search.MultiPhraseQuery", query
111                    .getClass().getName());
112    
113            query = new StandardQueryParser(new StandardAnalyzer(matchVersion)).parse("\"fox jumps\"", "content");
114            Assert.assertEquals(1, TestUtil.hitCount(searcher, query));
115    
116            Assert.assertEquals("content:\"fox jumps\"", query.toString());
117            Assert.assertEquals("org.apache.lucene.search.PhraseQuery", query
118                    .getClass().getName());
119    
120        }
121    
122        @Test
123        public void testTermQuery() throws CorruptIndexException, IOException,
124                QueryNodeException {
125    
126            Document doc = new Document();
127            doc.add(new Field("content", "I work for the united nations",
128                    TextField.TYPE_STORED));
129    
130            writer.addDocument(doc);
131    
132            searcher = new IndexSearcher(DirectoryReader.open(writer, false));
133    
134            StandardQueryParser parser = new StandardQueryParser(new SimpleAnalyzer(
135                    matchVersion));
136    
137            Query query = parser.parse("united nations", "content");
138    
139            Assert.assertEquals(1, TestUtil.hitCount(searcher, query));
140    
141        }
